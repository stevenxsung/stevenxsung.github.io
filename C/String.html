String

strlen: get the length of a string 
#include 
char str[64];
strcpy(str, “this is a test.”);
printf(“%d”, strlen(str)); /* get 15 */
strcmp: compare two strings 
#include 
strcmp(str1,str2)
/*
 * <0: str1 < str2
 * >0: str1 > str2
 * =0: str1 = str2
 * notice that if(-1)=true, if(1)=true, if(0)=false
 * that’s why we usually use if(!strcmp(str1,str2))
 */
strcpy: assign string to a character array and automatically append a ‘\0’ 
#include 
char str[3+1]; /* last one is for ‘\0’ */
strcpy(str, “abc”);
strcat: cast two strings together 
#include 
char str[64];
strcpy(str, “hello,”);
strcat(str, “this is a test.”);
strchr: locate the 1st occurrence of a character 
#include 
char str[] = “This is a test.”;
char *pointer;
pointer = strchr(str, ‘a’);
while(pointer!=NULL) {
	/*
	 * the target’s position = pointer-str+1
	 * suppose &str = 20, &pointer = 23,
	 * then the target character will be at 23-20+1=4th bit
	 */
}
strtok: split string into tokens 
char str[] = “this, is. a-test”;
char* pointer;
pointer = strtok(str, “ ,.-”);
while(pointer!=NULL) {
	/*
	 * you will get: “this”, “is”, “a”, “test” in the pointer
	 * all the characters in “ ,.-” will be eliminated
	 */
pointer = strtok(NULL, “ ,.-”);
}
