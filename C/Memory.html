IO

memset: set block of memory 
#include 
char str[1024];
memset(str, ‘\0’, sizeof(str));
memmove: move block of memory 
#include 
char str[] = “0123456789?!”;
memmove(str+5, str+10, 2);
/*
 * memmove(destination, source, length)
 * str is now “01234?!789?!”
 */
malloc: allocate memory block 
#include 
char* buffer = NULL;
buffer = (char*) malloc(10);
free(buffer);
realloc: changes the size of the memory block 
#include 
char* buffer = NULL;
buffer = (char*) realloc();
